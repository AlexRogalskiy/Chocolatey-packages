; This script will exit after 10 seconds
SetTimer, TimeOut, 10000

WinGet, id, ID, ahk_class TWizardForm, DirEdit
WinActivate, ahk_id %id%

winget, ActiveControlList, ControlList, ahk_id %id%

Loop, Parse, ActiveControlList, `n
{
  ControlGet, CanSee, Visible ,, %A_LoopField%, ahk_class TWizardForm
  ControlGet, CanWork, Enabled ,, %A_LoopField%, ahk_class TWizardForm
  ControlGet, Check, Checked ,, %A_LoopField%, ahk_class TWizardForm
  ControlGetText, RText, %A_LoopField%, ahk_class TWizardForm
  If (CanSee 
      and CanWork
      and (RText <> "")
      and InStr(A_LoopField,"radio",false)
      and !Check)
  {
    ControlClick, %A_LoopField%, ahk_id %id%
    Break
  }
}
sleep, 100
WinActivate, ahk_id %id%

; Next on license agreement page
Send, {enter}
sleep, 100
WinActivate, ahk_id %id%

; Next on destination page
Send, {enter}
sleep, 100
WinActivate, ahk_id %id%

; Next on components page
Send, {enter}
sleep, 100
WinActivate, ahk_id %id%

; Next on animal picture page
Send, {enter}

; Wait an unknown time
while (WinExist(ahk_id %id%)
sleep, 100
WinActivate, ahk_id %id%
Send, {enter}

ExitApp
